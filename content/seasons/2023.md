## Overview
### Background
On September 15, 2021, La Ligue was unanimously approved and sanctioned by the Canadian Soccer Association. Open tryouts were held in Montréal, Québec City, Gatineau, and Sherbrooke for players aged 15 and older. The province's first professional football league debuted with eighteen clubs, representing cities and regions across Quebec as well as one in Ontario.

  (function () {
    var table = document.getElementById("teams-table-2023");
    if (!table) return;

    var headers = table.querySelectorAll("thead th");
    var tbody = table.querySelector("tbody");

    function toComparable(value, isNumeric) {
      var text = value.trim();
      if (isNumeric) {
        return Number(text.replace(/[^\d.-]/g, "")) || 0;
      }
      return text.toLowerCase();
    }

    headers.forEach(function (header, index) {
      header.style.cursor = "pointer";
      header.title = "Sort";
      header.dataset.sortDirection = "asc";

      header.addEventListener("click", function () {
        var direction = header.dataset.sortDirection === "asc" ? "desc" : "asc";
        header.dataset.sortDirection = direction;

        var rows = Array.from(tbody.querySelectorAll("tr"));
        var isNumeric = index === 3;

        rows.sort(function (a, b) {
          var aValue = toComparable(a.children[index].textContent || "", isNumeric);
          var bValue = toComparable(b.children[index].textContent || "", isNumeric);

          if (aValue < bValue) return direction === "asc" ? -1 : 1;
          if (aValue > bValue) return direction === "asc" ? 1 : -1;
          return 0;
        });

        rows.forEach(function (row) {
          tbody.appendChild(row);
        });
      });
    });
  })();
</script>
