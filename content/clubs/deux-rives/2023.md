  (function () {
    var table = document.getElementById("dr-2023-squad");
    if (!table) return;

    var headers = table.querySelectorAll("thead th");
    var tbody = table.querySelector("tbody");

    function getCellText(row, index) {
      var cell = row.children[index];
      return (cell && cell.textContent ? cell.textContent : "").trim();
    }

    function toComparable(text, isNumeric) {
      if (isNumeric) {
        var num = Number(text.replace(/[^\d.-]/g, ""));
        return Number.isFinite(num) ? num : -Infinity;
      }
      return text.toLowerCase();
    }

    function sortWithinGroups(columnIndex, direction) {
      var rows = Array.from(tbody.querySelectorAll("tr"));

      var groups = [];
      var current = null;

      rows.forEach(function (row) {
        if (row.classList.contains("squad-group-row")) {
          current = { header: row, items: [] };
          groups.push(current);
        } else if (current) {
          current.items.push(row);
        }
      });

      if (groups.length === 0) return;

      var isNumeric = columnIndex === 0 || columnIndex === 6;

      groups.forEach(function (group) {
        group.items.sort(function (a, b) {
          var aValue = toComparable(getCellText(a, columnIndex), isNumeric);
          var bValue = toComparable(getCellText(b, columnIndex), isNumeric);
          if (aValue < bValue) return direction === "asc" ? -1 : 1;
          if (aValue > bValue) return direction === "asc" ? 1 : -1;
          return 0;
        });
      });

      groups.forEach(function (group) {
        tbody.appendChild(group.header);
        group.items.forEach(function (row) {
          tbody.appendChild(row);
        });
      });
    }

    headers.forEach(function (header, index) {
      header.style.cursor = "pointer";
      header.title = "Sort";
      header.dataset.sortDirection = "asc";

      header.addEventListener("click", function () {
        var direction = header.dataset.sortDirection === "asc" ? "desc" : "asc";
        header.dataset.sortDirection = direction;
        sortWithinGroups(index, direction);
      });
    });
  })();
